## API

```
function useEventListener<T extends Target = HTMLElement>(
  eventName: string,
  handler: Function,
  options?: { eventListenerOptions?: boolean | AddEventListenerOptions },
): MutableRefObject<T>;

function useEventListener<T extends Target = HTMLElement>(
  eventName: string,
  handler: Function,
  options?: { dom: Dom, eventListenerOptions?: boolean | AddEventListenerOptions },
): void

type Target = HTMLElement | Window;
type Options = { dom?: Dom, eventListenerOptions?: boolean | AddEventListenerOptions }
type Dom = Target | (() => Target) | null;

```

### Property

| Property    | Description     | type                   | default |
|---------|----------|------------------------|--------|
| eventName | event name | string | -      |
| handler | handle function | Function | -      |
| options | more options(optional) | Options |      |

### Options

| Property    | Description     | type                   | default |
|---------|----------|------------------------|--------|
| dom? | optional, if none is passed, this hook will subscribe to the ref that it returns	 | HTMLElement \| (() => HTMLElement) \| null   | Window      |
| eventListenerOptions? | optional, params for EventListener | boolean \| AddEventListenerOptions   |       |

### Return

| Property | Description                                                       | Type                 |
|------|----------|------|
| ref | When no param is passed, this ref will be listened | `RefObject<HTMLElement>` |
